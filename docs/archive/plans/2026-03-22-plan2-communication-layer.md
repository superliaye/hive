# Plan 2: Communication Layer — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the SQLite-backed communication layer (ICommsProvider), channel manager, message gateway with audit logging, and two CLI commands (`hive chat`, `hive observe`) so agents can post/read messages and the super user can talk to the CEO.

**Architecture:** A `SqliteCommsProvider` implements the `ICommsProvider` interface using better-sqlite3 with FTS5 for full-text search. A `ChannelManager` auto-generates channels from the org tree. A `MessageGateway` wraps the provider with audit logging. CLI commands use Commander.js subcommands wired to the gateway.

**Tech Stack:** TypeScript, Node.js 20+, Commander.js, better-sqlite3 (with FTS5), chalk, tsx (dev runner), vitest

**Spec:** `docs/specs/2026-03-22-hive-platform-design.md` — sections "Communication Layer — Canopy Integration" and "Super User Interface"

Create `src/comms/sqlite-provider.ts`:

```typescript
import Database from 'better-sqlite3';
import { randomUUID } from 'crypto';
import type { Message, Channel, ICommsProvider } from './types.js';

function extractMentions(content: string): string[] {
  const matches = content.match(/@([\w-]+)/g);
  return matches ? matches.map(m => m.slice(1)) : [];
}

function rowToMessage(row: any): Message {
  return {
    id: row.id,
    channel: row.channel,
    sender: row.sender,
    content: row.content,
    thread: row.thread ?? undefined,
    mentions: row.mentions ? JSON.parse(row.mentions) : undefined,
    timestamp: new Date(row.timestamp),
    metadata: row.metadata ? JSON.parse(row.metadata) : undefined,
  };
}

function rowToChannel(row: any, members: string[]): Channel {
  return {
    id: row.id,
    name: row.name,
    members,
    createdAt: new Date(row.created_at),
    autoGenerated: row.auto_generated === 1,
  };
}

export class SqliteCommsProvider implements ICommsProvider {
  private db: Database.Database;

  constructor(dbPath: string) {
    this.db = new Database(dbPath);
    this.db.pragma('journal_mode = WAL');
    this.db.pragma('foreign_keys = ON');
    this.migrate();
  }

  private migrate(): void {
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS channels (
        id TEXT PRIMARY KEY,
        name TEXT NOT NULL UNIQUE,
        auto_generated INTEGER NOT NULL DEFAULT 0,
        created_at DATETIME DEFAULT CURRENT_TIMESTAMP
      );

      CREATE TABLE IF NOT EXISTS channel_members (
        channel_id TEXT NOT NULL,
        agent_id TEXT NOT NULL,
        joined_at DATETIME DEFAULT CURRENT_TIMESTAMP,
        PRIMARY KEY (channel_id, agent_id),
        FOREIGN KEY (channel_id) REFERENCES channels(id)
      );

      CREATE TABLE IF NOT EXISTS messages (
        id TEXT PRIMARY KEY,
        channel TEXT NOT NULL,
        sender TEXT NOT NULL,
        content TEXT NOT NULL,
        thread TEXT,
        mentions TEXT,
        metadata TEXT,
        timestamp DATETIME DEFAULT (strftime('%Y-%m-%dT%H:%M:%f', 'now')),
        FOREIGN KEY (channel) REFERENCES channels(name)
      );
      CREATE INDEX IF NOT EXISTS idx_messages_channel ON messages(channel);
      CREATE INDEX IF NOT EXISTS idx_messages_sender ON messages(sender);
      CREATE INDEX IF NOT EXISTS idx_messages_timestamp ON messages(timestamp);
      CREATE INDEX IF NOT EXISTS idx_messages_thread ON messages(thread);

      CREATE TABLE IF NOT EXISTS read_receipts (
        agent_id TEXT NOT NULL,
        message_id TEXT NOT NULL,
        read_at DATETIME DEFAULT CURRENT_TIMESTAMP,
        PRIMARY KEY (agent_id, message_id),
        FOREIGN KEY (message_id) REFERENCES messages(id)
      );
    `);

    // FTS5 virtual table for full-text search on message content
    // Wrapped in try-catch because CREATE VIRTUAL TABLE IF NOT EXISTS
    // is not supported in older SQLite versions
    try {
      this.db.exec(`
        CREATE VIRTUAL TABLE IF NOT EXISTS messages_fts USING fts5(
          id UNINDEXED,
          channel UNINDEXED,
          sender UNINDEXED,
          content,
          content_rowid='rowid'
        );
      `);
    } catch {
      // FTS table already exists
    }

    // Triggers to keep FTS in sync with messages table
    this.db.exec(`
      CREATE TRIGGER IF NOT EXISTS messages_ai AFTER INSERT ON messages BEGIN
        INSERT INTO messages_fts(rowid, id, channel, sender, content)
        VALUES (NEW.rowid, NEW.id, NEW.channel, NEW.sender, NEW.content);
      END;

      CREATE TRIGGER IF NOT EXISTS messages_ad AFTER DELETE ON messages BEGIN
        INSERT INTO messages_fts(messages_fts, rowid, id, channel, sender, content)
        VALUES ('delete', OLD.rowid, OLD.id, OLD.channel, OLD.sender, OLD.content);
      END;
    `);
  }

  async createChannel(name: string, members?: string[]): Promise<Channel> {
    const existing = this.db.prepare('SELECT id FROM channels WHERE name = ?').get(name);
    if (existing) {
      throw new Error(`Channel already exists: ${name}`);
    }

    const id = randomUUID();
    this.db.prepare(
      'INSERT INTO channels (id, name, auto_generated) VALUES (?, ?, 0)',
    ).run(id, name);

    const memberList = members ?? [];
    const insertMember = this.db.prepare(
      'INSERT INTO channel_members (channel_id, agent_id) VALUES (?, ?)',
    );
    for (const agentId of memberList) {
      insertMember.run(id, agentId);
    }

    return {
      id,
      name,
      members: memberList,
      createdAt: new Date(),
      autoGenerated: false,
    };
  }

  async listChannels(): Promise<Channel[]> {
    const rows = this.db.prepare('SELECT * FROM channels ORDER BY name').all() as any[];
    return rows.map((row) => {
      const memberRows = this.db.prepare(
        'SELECT agent_id FROM channel_members WHERE channel_id = ?',
      ).all(row.id) as any[];
      return rowToChannel(row, memberRows.map((m: any) => m.agent_id));
    });
  }

  async postMessage(
    channel: string,
    sender: string,
    content: string,
    opts?: { thread?: string },
  ): Promise<Message> {
    // Verify channel exists
    const ch = this.db.prepare('SELECT id FROM channels WHERE name = ?').get(channel);
    if (!ch) {
      throw new Error(`Channel not found: ${channel}`);
    }

    const id = randomUUID();
    const thread = opts?.thread ?? null;
    const mentions = extractMentions(content);
    const mentionsJson = mentions.length > 0 ? JSON.stringify(mentions) : null;

    this.db.prepare(
      'INSERT INTO messages (id, channel, sender, content, thread, mentions) VALUES (?, ?, ?, ?, ?, ?)',
    ).run(id, channel, sender, content, thread, mentionsJson);

    // Read back the row to get the actual timestamp
    const row = this.db.prepare('SELECT * FROM messages WHERE id = ?').get(id) as any;
    return rowToMessage(row);
  }

  async readChannel(
    channel: string,
    opts?: { limit?: number; since?: Date },
  ): Promise<Message[]> {
    let sql: string;
    const params: unknown[] = [channel];

    if (opts?.limit && opts.limit > 0) {
      // When limit is specified, get the N most recent, then return in chronological order
      if (opts?.since) {
        sql = `
          SELECT * FROM (
            SELECT * FROM messages WHERE channel = ? AND timestamp > ? ORDER BY timestamp DESC LIMIT ?
          ) sub ORDER BY timestamp ASC
        `;
        params.push(opts.since.toISOString(), opts.limit);
      } else {
        sql = `
          SELECT * FROM (
            SELECT * FROM messages WHERE channel = ? ORDER BY timestamp DESC LIMIT ?
          ) sub ORDER BY timestamp ASC
        `;
        params.push(opts.limit);
      }
    } else if (opts?.since) {
      sql = 'SELECT * FROM messages WHERE channel = ? AND timestamp > ? ORDER BY timestamp ASC';
      params.push(opts.since.toISOString());
    } else {
      sql = 'SELECT * FROM messages WHERE channel = ? ORDER BY timestamp ASC';
    }

    const rows = this.db.prepare(sql).all(...params) as any[];
    return rows.map(rowToMessage);
  }

  async searchHistory(
    query: string,
    opts?: { channel?: string; sender?: string },
  ): Promise<Message[]> {
    // Use FTS5 for full-text search
    // Escape FTS5 special characters and add wildcard
    const ftsQuery = query.replace(/['"*()]/g, '').trim();
    if (!ftsQuery) return [];

    let sql = `
      SELECT m.* FROM messages m
      INNER JOIN messages_fts fts ON m.rowid = fts.rowid
      WHERE messages_fts MATCH ?
    `;
    const params: unknown[] = [`"${ftsQuery}"`];

    if (opts?.channel) {
      sql += ' AND m.channel = ?';
      params.push(opts.channel);
    }

    if (opts?.sender) {
      sql += ' AND m.sender = ?';
      params.push(opts.sender);
    }

    sql += ' ORDER BY m.timestamp DESC';

    const rows = this.db.prepare(sql).all(...params) as any[];
    return rows.map(rowToMessage);
  }

  async getUnread(agentId: string): Promise<Message[]> {
    // Get all messages in channels where agentId is a member,
    // excluding messages sent by the agent itself,
    // excluding messages already marked as read
    const sql = `
      SELECT m.* FROM messages m
      INNER JOIN channels c ON m.channel = c.name
      INNER JOIN channel_members cm ON c.id = cm.channel_id AND cm.agent_id = ?
      LEFT JOIN read_receipts rr ON m.id = rr.message_id AND rr.agent_id = ?
      WHERE m.sender != ?
        AND rr.message_id IS NULL
      ORDER BY m.timestamp ASC
    `;
    const rows = this.db.prepare(sql).all(agentId, agentId, agentId) as any[];
    return rows.map(rowToMessage);
  }

  async markRead(agentId: string, messageIds: string[]): Promise<void> {
    const stmt = this.db.prepare(
      'INSERT OR IGNORE INTO read_receipts (agent_id, message_id) VALUES (?, ?)',
    );
    const txn = this.db.transaction((ids: string[]) => {
      for (const msgId of ids) {
        stmt.run(agentId, msgId);
      }
    });
    txn(messageIds);
  }

  /**
   * Add a member to an existing channel. Used by ChannelManager during sync.
   */
  async addMember(channelName: string, agentId: string): Promise<void> {
    const ch = this.db.prepare('SELECT id FROM channels WHERE name = ?').get(channelName) as any;
    if (!ch) throw new Error(`Channel not found: ${channelName}`);
    this.db.prepare(
      'INSERT OR IGNORE INTO channel_members (channel_id, agent_id) VALUES (?, ?)',
    ).run(ch.id, agentId);
  }

  /**
   * Create a channel with the autoGenerated flag set. Used by ChannelManager.
   */
  async createAutoChannel(name: string, members: string[]): Promise<Channel> {
    const existing = this.db.prepare('SELECT id FROM channels WHERE name = ?').get(name);
    if (existing) {
      // Already exists — just ensure members are synced
      for (const agentId of members) {
        await this.addMember(name, agentId);
      }
      const ch = existing as any;
      const memberRows = this.db.prepare(
        'SELECT agent_id FROM channel_members WHERE channel_id = ?',
      ).all(ch.id) as any[];
      return {
        id: ch.id,
        name,
        members: memberRows.map((m: any) => m.agent_id),
        createdAt: new Date(),
        autoGenerated: true,
      };
    }

    const id = randomUUID();
    this.db.prepare(
      'INSERT INTO channels (id, name, auto_generated) VALUES (?, ?, 1)',
    ).run(id, name);

    const insertMember = this.db.prepare(
      'INSERT INTO channel_members (channel_id, agent_id) VALUES (?, ?)',
    );
    for (const agentId of members) {
      insertMember.run(id, agentId);
    }

    return {
      id,
      name,
      members,
      createdAt: new Date(),
      autoGenerated: true,
    };
  }

  close(): void {
    this.db.close();
  }
}
```

```typescript
import type { OrgChart } from '../types.js';
import type { Channel, ICommsProvider } from './types.js';

export class ChannelManager {
  constructor(private provider: ICommsProvider) {}

  /**
   * Sync channels from an OrgChart. Creates auto-generated channels
   * (all-hands, board, leadership, approvals, team channels) and adds
   * the correct members. Idempotent — safe to call repeatedly.
   */
  async syncFromOrgTree(org: OrgChart): Promise<Channel[]> {
    const created: Channel[] = [];

    for (const channelDef of org.channels) {
      const channel = await this.provider.createAutoChannel(
        channelDef.name,
        channelDef.memberIds,
      );
      created.push(channel);
    }

    return created;
  }

  /**
   * Ensure a channel exists with the given members. If it already exists,
   * members are added (not removed). Used for agent-created organic channels.
   */
  async ensureChannel(name: string, members?: string[]): Promise<Channel> {
    return this.provider.createAutoChannel(name, members ?? []);
  }

  /**
   * Get all channels that an agent is a member of.
   */
  async getChannelsForAgent(agentId: string): Promise<Channel[]> {
    const allChannels = await this.provider.listChannels();
    return allChannels.filter(c => c.members.includes(agentId));
  }
}
```

```typescript
import type { Message, Channel, ICommsProvider } from './types.js';
import type { AuditStore } from '../audit/store.js';

/**
 * MessageGateway wraps an ICommsProvider with audit logging.
 * This is the primary entry point for agents to send and receive messages.
 * Every postMessage call is logged to the audit store for traceability.
 */
export class MessageGateway {
  constructor(
    private provider: ICommsProvider,
    private auditStore: AuditStore,
  ) {}

  async postMessage(
    channel: string,
    sender: string,
    content: string,
    opts?: { thread?: string },
  ): Promise<Message> {
    const msg = await this.provider.postMessage(channel, sender, content, opts);

    // Log to audit trail
    this.auditStore.logInvocation({
      agentId: sender,
      invocationType: 'comms',
      model: 'n/a',
      inputSummary: `Posted to #${channel}: ${content.slice(0, 100)}`,
      outputSummary: `Message ${msg.id}`,
      channel,
    });

    return msg;
  }

  async readChannel(
    channel: string,
    opts?: { limit?: number; since?: Date },
  ): Promise<Message[]> {
    return this.provider.readChannel(channel, opts);
  }

  async searchHistory(
    query: string,
    opts?: { channel?: string; sender?: string },
  ): Promise<Message[]> {
    return this.provider.searchHistory(query, opts);
  }

  async listChannels(): Promise<Channel[]> {
    return this.provider.listChannels();
  }

  async getUnreadForAgent(agentId: string): Promise<Message[]> {
    return this.provider.getUnread(agentId);
  }

  async markRead(agentId: string, messageIds: string[]): Promise<void> {
    return this.provider.markRead(agentId, messageIds);
  }
}
```

```typescript
import type { MessageGateway } from './message-gateway.js';
import type { SqliteCommsProvider } from './sqlite-provider.js';
import type { Message } from './types.js';
import { spawnClaude, buildClaudeArgs } from '../agents/spawner.js';
import { assemblePrompt } from '../agents/prompt-assembler.js';
import { loadAgentConfig } from '../agents/config-loader.js';
import chalk from 'chalk';

export interface ChatActionOpts {
  message: string;
  gateway: MessageGateway;
  provider: SqliteCommsProvider;
  /** Path to the CEO agent directory. Required for CEO response. */
  ceoDir?: string;
  /** Skip spawning Claude CLI for CEO response (for testing). */
  skipCeoResponse?: boolean;
}

export interface ChatActionResult {
  userMessage: Message;
  ceoResponse?: Message;
}

/**
 * The `hive chat` action. Posts a message to #board as super-user,
 * then spawns the CEO agent via Claude CLI to get a response.
 */
export async function chatAction(opts: ChatActionOpts): Promise<ChatActionResult> {
  const { message, gateway, provider, ceoDir, skipCeoResponse } = opts;

  // Post the super user's message to #board
  const userMessage = await gateway.postMessage('board', 'super-user', message);

  if (skipCeoResponse || !ceoDir) {
    return { userMessage };
  }

  // Load CEO config and assemble prompt
  const ceoConfig = await loadAgentConfig(ceoDir, 'ceo', 0, null);
  const systemPrompt = assemblePrompt(ceoConfig);

  // Build context: recent #board history for the CEO
  const recentMessages = await provider.readChannel('board', { limit: 20 });
  const conversationContext = recentMessages
    .map(m => `[${m.timestamp.toISOString()}] ${m.sender}: ${m.content}`)
    .join('\n');

  const input = [
    'You have a new message from the super user on #board.',
    'Here is the recent conversation history:',
    '',
    conversationContext,
    '',
    'Respond to the super user\'s latest message. Be concise and direct.',
  ].join('\n');

  // Spawn Claude CLI as the CEO
  const args = buildClaudeArgs({
    model: ceoConfig.identity.model,
    systemPrompt,
    tools: ceoConfig.identity.tools,
  });

  const result = await spawnClaude(args, {
    cwd: ceoDir,
    input,
    timeoutMs: 120_000,
  });

  let ceoResponse: Message | undefined;
  if (result.exitCode === 0 && result.stdout.trim()) {
    ceoResponse = await gateway.postMessage('board', 'ceo', result.stdout.trim());
  }

  return { userMessage, ceoResponse };
}

export interface ObserveActionOpts {
  channel: string;
  gateway: MessageGateway;
  follow: boolean;
  limit: number;
  /** Callback for each new message in follow mode. */
  onMessage?: (formatted: string) => void;
}

export interface ObserveActionResult {
  messages: Message[];
  formatted: string;
}

/**
 * Format a single message for terminal display.
 */
export function formatMessage(msg: Message): string {
  const ts = msg.timestamp.toISOString().replace('T', ' ').slice(0, 19);
  const sender = chalk.bold.cyan(msg.sender);
  const thread = msg.thread ? chalk.dim(` (thread: ${msg.thread.slice(0, 8)}...)`) : '';
  return `${chalk.dim(ts)} ${sender}${thread}: ${msg.content}`;
}

/**
 * The `hive observe` action. Reads a channel's messages and optionally
 * tails new messages in real-time (poll-based).
 */
export async function observeAction(opts: ObserveActionOpts): Promise<ObserveActionResult> {
  const { channel, gateway, follow, limit } = opts;

  const messages = await gateway.readChannel(channel, { limit });
  const formatted = messages.map(formatMessage).join('\n');

  return { messages, formatted };
}

/**
 * Follow mode: polls for new messages and calls onMessage for each.
 * Returns an AbortController to stop polling.
 */
export function startFollowing(
  channel: string,
  gateway: MessageGateway,
  onMessage: (formatted: string) => void,
  pollIntervalMs: number = 1000,
): AbortController {
  const controller = new AbortController();
  let lastTimestamp = new Date();

  const poll = async () => {
    if (controller.signal.aborted) return;

    try {
      const newMessages = await gateway.readChannel(channel, { since: lastTimestamp });
      for (const msg of newMessages) {
        onMessage(formatMessage(msg));
        if (msg.timestamp > lastTimestamp) {
          lastTimestamp = msg.timestamp;
        }
      }
    } catch {
      // Channel may not exist yet — ignore errors in follow mode
    }

    if (!controller.signal.aborted) {
      setTimeout(poll, pollIntervalMs);
    }
  };

  // Start first poll
  setTimeout(poll, pollIntervalMs);

  return controller;
}
```

1. CLI parses command, opens `data/comms.db`
2. ChannelManager syncs channels from org tree (ensures #board exists)
3. Gateway posts `super-user` message to `#board`
4. Loads CEO agent config from org tree
5. Spawns `claude -p` with CEO's system prompt + #board conversation history
6. Posts CEO's response back to `#board`
7. Displays response to terminal

**Data flow for `hive observe board -f`:**
1. CLI opens `data/comms.db`
2. Reads last N messages from `#board`
3. Formats with timestamps + sender names + content
4. In follow mode: polls every 1s for new messages, prints as they arrive
5. Ctrl+C gracefully stops polling
